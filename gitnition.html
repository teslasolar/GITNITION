⚙️ Setup — host this page on your gateway

1. Create a WebDev Python resource

In Designer → Project Browser → WebDev → right-click → New Python Resource. Name it gitnition. Mount at path gitnition. Allow GET + POST.

2. Paste the Python below into the resource

 

# GITNITION — Ignition WebDev Python resource # Drop this into a WebDev project as a Python resource at endpoint "gitnition". # Then point your browser at: # http://<gateway>:8088/system/webdev/<projectName>/gitnition # # Configure mount point in WebDev resource settings: # - Mount path: gitnition # - Allow GET, POST # - Authentication: as required for your environment # # This file handles GET/POST for the same endpoint; route by request["servletRequest"].getMethod() # and by query parameter "op" or by JSON body "op" key. import json, system INDEX_HTML = """__INDEX_PLACEHOLDER__""" # ← paste the HTML body here, or fetch from WebDev static def _json(obj, status=200): return {"json": obj, "response": status} def _err(msg, status=400): return _json({"error": str(msg)}, status) def doGet(request, session): p = request.get("params", {}) op = p.get("op", "index") if op == "index": # Serve the HTML page itself return {"html": INDEX_HTML} if op == "browse": path = p.get("path", "") try: results = system.tag.browse(path=path, filter={"recursive": False}) out = [] for r in results.getResults(): out.append({ "name": str(r.get("name", "")), "fullPath": str(r.get("fullPath", "")), "tagType": str(r.get("tagType", "")), "dataType": str(r.get("dataType", "")) if r.get("dataType") else None, "hasChildren": bool(r.get("hasChildren", False)), "udtType": str(r.get("typeId", "")) if r.get("typeId") else None, }) return _json({"path": path, "results": out, "total": len(out)}) except Exception as e: return _err(e) if op == "read": paths_csv = p.get("paths", "") if not paths_csv: return _err("paths required") paths = [x for x in paths_csv.split(",") if x] try: qvs = system.tag.readBlocking(paths) out = [] for path, qv in zip(paths, qvs): out.append({ "path": path, "value": qv.value, "quality": str(qv.quality), "timestamp": str(qv.timestamp), "good": qv.quality.isGood(), }) return _json({"results": out}) except Exception as e: return _err(e) if op == "udts": provider = p.get("provider", "default") try: results = system.tag.browse(path="[%s]_types_" % provider, filter={"tagType": "UdtType", "recursive": True}) out = [] for r in results.getResults(): out.append({"name": str(r.get("name", "")), "fullPath": str(r.get("fullPath", "")), "typeId": str(r.get("fullPath", "")).replace("[%s]_types_/" % provider, "")}) return _json({"udts": out, "total": len(out)}) except Exception as e: return _err(e) if op == "history": path = p.get("path", "") if not path: return _err("path required") range_h = int(p.get("hours", "1")) try: ds = system.tag.queryTagHistory( paths=[path], startDate=system.date.addHours(system.date.now(), -range_h), endDate=system.date.now(), returnSize=200, aggregationMode="LastValue", returnFormat="Wide", ) rows = [] for r in range(ds.rowCount): rows.append({ "ts": str(ds.getValueAt(r, 0)), "v": ds.getValueAt(r, 1), }) return _json({"path": path, "rows": rows}) except Exception as e: return _err(e) if op == "alarms": try: alarms = system.alarm.queryStatus(state=["ActiveUnacked", "ActiveAcked"]) out = [] for a in alarms: out.append({ "uuid": str(a.getId()), "source": str(a.getSource()), "displayPath": str(a.getDisplayPath()), "priority": str(a.getPriority()), "state": str(a.getState()), "name": str(a.getName()), "active": a.isActive(), "acked": a.isAcked(), }) return _json({"alarms": out, "total": len(out)}) except Exception as e: return _err(e) if op == "version": return _json({"version": str(system.util.getGatewayInfo() if hasattr(system.util, "getGatewayInfo") else "unknown"), "gitnition": "1.0"}) return _err("unknown op: %s" % op, 404) def doPost(request, session): body = {} try: body = json.loads(request.get("postData", "") or "{}") except Exception as e: return _err("bad JSON: %s" % e) op = body.get("op", "") if op == "write": # body: { op:"write", writes:[{path, value},...] values) out = [] for w, qc in zip(writes, qcs): out.append({"path": w["path"], "ok": qc.isGood(), "quality": str(qc)}) return _json({"results": out}) except Exception as e: return _err(e) if op == "instantiate": # body: { op:"instantiate", typeId:"Generic/Devices/Pump", path:"My/Folder/P-101", params:{...} } typeId = body.get("typeId", "") path = body.get("path", "") params = body.get("params", {}) provider = body.get("provider", "default") if not typeId or not path: return _err("typeId+path required") try: collisionPolicy = body.get("collisionPolicy", "a") # a=abort, o=overwrite, m=merge, i=ignore tag = { "name": path.split("/")[-1], "tagType": "UdtInstance", "typeId": typeId, "parameters": {k: {"value": v} for k, v in params.items()}, } parent = "[%s]" % provider + "/".join(path.split("/")[:-1]) r = system.tag.configure(parent, [tag], collisionPolicy) return _json({"results": [str(x) for x in r], "createdAt": parent + "/" + tag["name"]}) except Exception as e: return _err(e) if op == "import": # body: { op:"import", json: <udt-defs-json>, provider:"default" } provider = body.get("provider", "default") udt_json = body.get("json", None) if not udt_json: "seconds": secs}) except Exception as e: return _err(e) return _err("unknown op: %s" % op, 404)

3. Save the index HTML

The Python references INDEX_HTML. You can either (a) paste the entire HTML body of this page into that variable, or (b) host the HTML separately as a WebDev static resource and have the Python redirect to it. Easiest path is option (b) — drop this very file into data/projects/<projectName>/webdev/<mountPath> as a static file.

4. Hit the endpoint

http://<gateway>:8088/system/webdev/<projectName>/gitnition

The page auto-detects same-origin and starts talking. If you load it from elsewhere, set the Base URL in the header bar.

Status & permissions

Auth
Browser session cookie if same-origin. Otherwise basic/apikey/bearer prompt.
Network
Same-origin = no CORS. Cross-origin needs Access-Control-Allow-Origin on the gateway.
Project scope
WebDev runs in the project context — only tag providers visible to that project are reachable.
Roles
Recommend role-gating the WebDev resource: read-only role for browse/read/history, admin role for write/instantiate/import.
